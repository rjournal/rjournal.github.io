\begin{algorithm}
\begin{algorithmic}[H]
\Procedure{MatchW}{data}
\State find $W_i$ for each $i$ using \code{Match} function
\State $M \coloneqq \sum_i W_i$ %\Comment{find matched within}
\State $mdata\coloneqq data[M]$ \Comment{extracts matched data}
\If {data contains outcome variable $Y$}:
\State estimate $\widehat{ATT}$ and $sd(\widehat{ATT})$ from model on $mdata$
\Else
\State $\widehat{ATT}$<-$sd(\widehat{ATT})$<-NULL
\EndIf
\State \textbf{return} $mdata$, $\widehat{ATT}$ and $sd(\widehat{ATT})$ %\Comment{returns matched dataset}
\EndProcedure
\end{algorithmic}
\end{algorithm}
