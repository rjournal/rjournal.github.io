\begin{algorithm}
	\caption{Calculation of $(I-A)^{-1}D^{1/2}$}
	\label{alg:approxchol}
	\begin{algorithmic}
		\State Let $\Tilde{L}$ be an $n \times n$ lower triangular matrix
		\ForAll{$i \in 1,...,n$}
		\ForAll{$k \in i,...,n$}
		\State Let $l = A_{\mathcal{N}_i,i}^T \Tilde{L}_{\mathcal{N}_i,k}$
		\If{$i=k$}
		\State $\Tilde{L}_{i,k} = (1 + l)\sqrt{D_{kk}}$
		\Else
		\State $\Tilde{L}_{i,k} = l\sqrt{D_{kk}}$
		\EndIf
		\EndFor
		\EndFor
	\end{algorithmic}
\end{algorithm}
