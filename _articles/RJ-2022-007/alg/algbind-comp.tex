  \SetKwFunction{FSuccessful}{successful}
  \SetKwProg{Fn}{function}{:}{}
  \Fn{\Fmain{x, g}}{
    \FIf{\FSuccessful{x}}{
        $y$ = \FExtract{x} \;
        $z$ = $g$($y$) \;
        \Return{z} \;
    } \FElse {} {
        \Return{$fail$} \;
    }
  }
