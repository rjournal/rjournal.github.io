\begin{algorithm}[H]
		\Procedure{AggregateCellScores}{}
		\State Initialize $S_{t,w,h} \gets 0 , \forall t\in [1,4], w,h \in [1,N_A-1]$
		
		\For{$i1 = 0$ to $N_A - 1$}
		\For{$i2 = i1 + 1$ to $N_A$}
		\State{$w_0 \gets w(i2,i1)$}
		\For{$j1 = 0$ to $N_A - 1$}
		\For{$j2 = j1+1 $ to $N_A$}
		\State{$t \gets Type(i1,i2,j1,j2)$}
		\State{$h_0 \gets h(j2,j1)$}
		\State{$o \gets O(i1,i2,j1,j2)$}
		\State{$e \gets E(i1,i2,j1,j2)$}
		\State{$S_{t,w_0,h_0} \gets S_{t,w_0,h_0} + o\cdot ln\left(\frac{o}{e}\right)$}
		\EndFor
		\EndFor
		\EndFor
		\EndFor
		\EndProcedure
		\label{alg:HHG_Alg2}
	\end{algorithm}
