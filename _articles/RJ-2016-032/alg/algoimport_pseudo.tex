\begin{algorithm}[H]

  \KwIn{a data set containing \texttt{MAF} or \texttt{GISTIC} scores
    (e.g.,\ as obtained from cBio portal) (Cerami et al. (2012); Beroukhim et al. (2007)).}
  \KwResult{a data structure containing Boolean flags for ``events,''
    relative frequencies and other metadata.}
  \BlankLine
  From the data set (depending on the data format) derive a Boolean
  matrix $M$, where each entry $\langle i,\, j \rangle$ is \textsf{true}
  if event $i$ is ``present'' in sample/patient $j$.\;
  \BlankLine
  \ForAll{events $e$}{Compute the \emph{frequency} of the event $e$ in
    the data set and save it in a map $F$.\;
    Compute the \emph{joint probability} of co-occurrence of pair of events in
    the data set and save it in a map $C$.\;}
  \BlankLine
  \KwRet{A data structure comprising the Boolean matrix $M$, the
    maps $F$ and $C$ and other metadata.}
  \BlankLine
\end{algorithm}
