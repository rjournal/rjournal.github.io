\begin{algorithm}
\begin{algorithmic}[H]
\Procedure{MatchPW}{data}
\State find $W_i$ and $\overline{W}_i$ for each $i$ using \code{Match} function
\State $B\coloneqq \textrm{\code{Match}}(\sum_i \overline{W}_i \cup \textrm{all controls}$)%\Comment{find matched between}
\State $M\coloneqq \sum_i W_i + B$
\State $mdata\coloneqq data[M]$\Comment{extracts matched data}
\If{data contains outcome variable $Y$:}
\State estimate $\widehat{ATT}$ and $sd(\widehat{ATT})$ from model on $mdata$
\Else
\State $\widehat{ATT}$<-$sd(\widehat{ATT})$<-NULL
\EndIf
\State \textbf{return} $mdata$, $\widehat{ATT}$ and $sd(\widehat{ATT})$  %\Comment{returns matched dataset, ATT}
\EndProcedure
\end{algorithmic}
\end{algorithm}
